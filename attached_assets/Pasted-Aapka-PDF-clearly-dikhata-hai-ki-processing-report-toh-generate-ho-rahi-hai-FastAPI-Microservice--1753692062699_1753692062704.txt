ðŸ›  Step 2: FastAPI Route for File Upload + Merge + Download

from fastapi import FastAPI, UploadFile, File
from fastapi.responses import FileResponse
import shutil
import os
from PyPDF2 import PdfMerger

app = FastAPI()

@app.post("/merge-pdf/")
async def merge_pdfs(files: list[UploadFile] = File(...)):
    output_filename = "merged_result.pdf"
    temp_files = []

    # Save uploaded files temporarily
    for file in files:
        temp_path = f"temp_{file.filename}"
        with open(temp_path, "wb") as buffer:
            shutil.copyfileobj(file.file, buffer)
        temp_files.append(temp_path)

    # Merge using PyPDF2
    merger = PdfMerger()
    for f in temp_files:
        merger.append(f)
    merger.write(output_filename)
    merger.close()

    # Cleanup temp files
    for f in temp_files:
        os.remove(f)

    # Return merged PDF file as download
    return FileResponse(path=output_filename, media_type='application/pdf', filename=output_filename)
